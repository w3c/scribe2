632 $content =~ s/([^\12\040-\176])/sprintf('\\x{%X}',ord($1))/eg;
633

636 $content =~ s/(\n+)\n/("\\n" x length($1)) . "\n"/eg;

640 $no_content = "(no content)" unless defined $no_content;
641 if ($content eq $no_content) {
642 # escape our $no_content marker

644 }
645 elsif ($content eq "") {
646 $content = $no_content;
647 }
648 }
649
650 my @dump;
651 push(@dump, $opt{preheader}) if $opt{preheader};
652 push(@dump, $self->{_headers}->as_string, $content);
653 push(@dump, "(+ $chopped more bytes not shown)") if $chopped;
654
655 my $dump = join("\n", @dump, "");
656 $dump =~ s/^/$opt{prefix}/gm if $opt{prefix};
657
658 print $dump unless defined wantarray;
659 return $dump;

775sub _parts {
776 my $self = shift;
777 my $ct = $self->content_type;
778 if ($ct =~ m,^multipart/,) {
779 require HTTP::Headers::Util;
780 my @h = HTTP::Headers::Util::split_header_words($self->header("Content-Type"));
781 die "Assert" unless @h;
782 my %h = @{$h[0]};
783 if (defined(my $b = $h{boundary})) {
784 my $str = $self->content;
785 $str =~ s/\r?\n--\Q$b\E--.*//s;
786 if ($str =~ s/(^|.*?\r?\n)--\Q$b\E\r?\n//s) {
787 $self->{_parts} = [map $self->_part_class->parse($_),
788 split(/\r?\n--\Q$b\E\r?\n/, $str)]
789 }
790 }
791 }
792 elsif ($ct eq "message/http") {
793 require HTTP::Request;
794 require HTTP::Response;
795 my $content = $self->content;
796 my $class = ($content =~ m,^(HTTP/.*)\n,) ?
797 "HTTP::Response" : "HTTP::Request";
798 $self->{_parts} = [$class->parse($content)];
799 }
800 elsif ($ct =~ m,^message/,) {
801 $self->{_parts} = [ $self->_part_class->parse($self->content) ];
802 }

809sub _content {
810 my $self = shift;
811 my $ct = $self->{_headers}->header("Content-Type") || "multipart/mixed";
812 if ($ct =~ m,^\s*message/,i) {
813 _set_content($self, $self->{_parts}[0]->as_string($CRLF), 1);
814 return;
815 }
816
817 require HTTP::Headers::Util;
818 my @v = HTTP::Headers::Util::split_header_words($ct);
819 Carp::carp("Multiple Content-Type headers") if @v > 1;
820 @v = @{$v[0]};
821
822 my $boundary;
823 my $boundary_index;
824 for (my @tmp = @v; @tmp;) {
825 my($k, $v) = splice(@tmp, 0, 2);
826 if ($k eq "boundary") {
827 $boundary = $v;
828 $boundary_index = @v - @tmp - 1;
829 last;
830 }
831 }
832
833 my @parts = map $_->as_string($CRLF), @{$self->{_parts}};
834
835 my $bno = 0;
836 $boundary = _boundary() unless defined $boundary;
837 CHECK_BOUNDARY:
838 {
839 for (@parts) {
840 if (index($_, $boundary) >= 0) {
841 # must have a better boundary
842 $boundary = _boundary(++$bno);
843 redo CHECK_BOUNDARY;
844 }
845 }
846 }
847
848 if ($boundary_index) {
849 $v[$boundary_index] = $boundary;
850 }
851 else {
852 push(@v, boundary => $boundary);
853 }
854
855 $ct = HTTP::Headers::Util::join_header_words(@v);
856 $self->{_headers}->header("Content-Type", $ct);
857
858 _set_content($self, "--$boundary$CRLF" .
859 join("$CRLF--$boundary$CRLF", @parts) .
860 "$CRLF--$boundary--$CRLF",
861 1);
862}

882

890
891 use parent 'HTTP::Message';

893=head1 DESCRIPTION
894
895An C<HTTP::Message> object contains some headers and a content body.
896The following methods are available:
897
898=over 4
899
900=item $mess = HTTP::Message->new
901
902=item $mess = HTTP::Message->new( $headers )
903
904=item $mess = HTTP::Message->new( $headers, $content )
905
906This constructs a new message object. Normally you would want
907construct C<HTTP::Request> or C<HTTP::Response> objects instead.
908
909The optional $header argument should be a reference to an
910C<HTTP::Headers> object or a plain array reference of key/value pairs.
911If an C<HTTP::Headers> object is provided then a copy of it will be
912embedded into the constructed message, i.e. it will not be owned and
913can be modified afterwards without affecting the message.
914
915The optional $content argument should be a string of bytes.
916
917=item $mess = HTTP::Message->parse( $str )
918
919This constructs a new message object by parsing the given string.
920
921=item $mess->headers
922
923Returns the embedded C<HTTP::Headers> object.
924
925=item $mess->headers_as_string
926
927=item $mess->headers_as_string( $eol )
928
929Call the as_string() method for the headers in the
930message. This will be the same as
931
932 $mess->headers->as_string
933
934but it will make your program a whole character shorter :-)
935
936=item $mess->content
937
938=item $mess->content( $bytes )
939
940The content() method sets the raw content if an argument is given. If no
941argument is given the content is not touched. In either case the
942original raw content is returned.
943
944If the C<undef> argument is given, the content is reset to its default value,
945which is an empty string.
946
947Note that the content should be a string of bytes. Strings in perl
948can contain characters outside the range of a byte. The C<Encode>
949module can be used to turn such strings into a string of bytes.
950
951=item $mess->add_content( $bytes )
952
953The add_content() methods appends more data bytes to the end of the
954current content buffer.
955
956=item $mess->add_content_utf8( $string )
957
958The add_content_utf8() method appends the UTF-8 bytes representing the
959string to the end of the current content buffer.
960
961=item $mess->content_ref
962
963=item $mess->content_ref( \$bytes )
964
965The content_ref() method will return a reference to content buffer string.
966It can be more efficient to access the content this way if the content
967is huge, and it can even be used for direct manipulation of the content,
968for instance:
969
970 ${$res->content_ref} =~ s/\bfoo\b/bar/g;
971
972This example would modify the content buffer in-place.
973
974If an argument is passed it will setup the content to reference some
975external source. The content() and add_content() methods
976will automatically dereference scalar references passed this way. For
977other references content() will return the reference itself and
978add_content() will refuse to do anything.
979
980=item $mess->content_charset
981
982This returns the charset used by the content in the message. The
983charset is either found as the charset attribute of the
984C<Content-Type> header or by guessing.
985
986See L<http://www.w3.org/TR/REC-html40/charset.html#spec-char-encoding>
987for details about how charset is determined.
988
989=item $mess->decoded_content( %options )
990
991Returns the content with any C<Content-Encoding> undone and, for textual content
992(C<Content-Type> values starting with C<text/>, exactly matching
993C<application/xml>, or ending with C<+xml>), the raw content's character set
994decoded into Perl's Unicode string format. Note that this
995L<does not currently|https://github.com/libwww-perl/HTTP-Message/pull/99>
996attempt to decode declared character sets for any other content types like
997C<application/json> or C<application/javascript>. If the C<Content-Encoding>
998or C<charset> of the message is unknown, this method will fail by returning
999C<undef>.
1000
1001The following options can be specified.
1002
1003=over
1004
1005=item C<charset>
1006
1007This overrides the charset parameter for text content. The value
1008C<none> can used to suppress decoding of the charset.
1009
1010=item C<default_charset>
1011
1012This overrides the default charset guessed by content_charset() or
1013if that fails "ISO-8859-1".
1014
1015=item C<alt_charset>
1016
1017If decoding fails because the charset specified in the Content-Type header
1018isn't recognized by Perl's Encode module, then try decoding using this charset
1019instead of failing. The C<alt_charset> might be specified as C<none> to simply
1020return the string without any decoding of charset as alternative.
1021
1022=item C<charset_strict>
1023
1024Abort decoding if malformed characters is found in the content. By
1025default you get the substitution character ("\x{FFFD}") in place of
1026malformed characters.
1027
1028=item C<raise_error>
1029
1030If TRUE then raise an exception if not able to decode content. Reason
1031might be that the specified C<Content-Encoding> or C<charset> is not
1032supported. If this option is FALSE, then decoded_content() will return
1033C<undef> on errors, but will still set $@.
1034
1035=item C<ref>
1036
1037If TRUE then a reference to decoded content is returned. This might
1038be more efficient in cases where the decoded content is identical to
1039the raw content as no data copying is required in this case.
1040
1041=back

1043=item $mess->decodable
1044
1045=item HTTP::Message::decodable()
1046
1047This returns the encoding identifiers that decoded_content() can
1048process. In scalar context returns a comma separated string of
1049identifiers.
1050
1051This value is suitable for initializing the C<Accept-Encoding> request
1052header field.
1053
1054=item $mess->decode
1055
1056This method tries to replace the content of the message with the
1057decoded version and removes the C<Content-Encoding> header. Returns
1058TRUE if successful and FALSE if not.
1059
1060If the message does not have a C<Content-Encoding> header this method

1062
1063Note that the content of the message is still bytes after this method

1066
1067=item $mess->encode( $encoding, ... )
1068
1069Apply the given encodings to the content of the message. Returns TRUE
1070if successful. The "identity" (non-)encoding is always supported; other
1071currently supported encodings, subject to availability of required
1072additional modules, are "gzip", "deflate", "x-bzip2", "base64" and "br".
1073
1074A successful call to this function will set the C<Content-Encoding>
1075header.
1076
1077Note that C<multipart/*> or C<message/*> messages can't be encoded and
1078this method will croak if you try.

1080=item $mess->parts
1081
1082=item $mess->parts( @parts )
1083
1084=item $mess->parts( \@parts )
1085
1086Messages can be composite, i.e. contain other messages. The composite
1087messages have a content type of C<multipart/*> or C<message/*>. This
1088method give access to the contained messages.
1089
1090The argumentless form will return a list of C<HTTP::Message> objects.
1091If the content type of $msg is not C<multipart/*> or C<message/*> then
1092this will return the empty list. In scalar context only the first
1093object is returned. The returned message parts should be regarded as
1094read-only (future versions of this library might make it possible
1095to modify the parent by modifying the parts).
1096
1097If the content type of $msg is C<message/*> then there will only be
1098one part returned.
1099
1100If the content type is C<message/http>, then the return value will be
1101either an C<HTTP::Request> or an C<HTTP::Response> object.
1102

1104modified. The array reference form is provided so that an empty list
1105can be provided. The @parts array should contain C<HTTP::Message>
1106objects. The @parts objects are owned by $mess after this call and
1107should not be modified or made part of other messages.
